1. Project Overview & Vision
This guide will walk you through building a jaw-dropping 3D portfolio website from scratch. The final result will feature immersive scroll-triggered animations, floating 3D geometry, cinematic particle backgrounds, smooth camera movements, and a dark space-like aesthetic — all using completely free libraries and assets.

1.1  What You Will Build
A full-page 3D scene powered by Three.js with rotating geometric objects
GSAP ScrollTrigger animations that fire as the user scrolls down
Parallax background images sourced from Unsplash & NASA APIs
Glitchy hero text with CSS clip-path and neon glow effects
Smooth section transitions using locomotive-scroll
Interactive project cards that tilt in 3D on mouse hover (VanillaTilt.js)
A floating animated navigation bar with blur-glass morphism style
Contact section with animated input fields
Fully responsive — works on desktop, tablet, and mobile

5. Three.js — 3D Background Scene (three-scene.js)
This is the most powerful piece of the site. The Three.js canvas sits behind all content and shows a living 3D world. Copy the complete code below into js/three-scene.js.
// ─── js/three-scene.js ───────────────────────────────
const canvas    = document.getElementById('three-canvas');
const renderer  = new THREE.WebGLRenderer({ canvas, alpha: true, antialias: true });
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
renderer.setSize(window.innerWidth, window.innerHeight);

const scene  = new THREE.Scene();
const camera = new THREE.PerspectiveCamera(75, window.innerWidth / window.innerHeight, 0.1, 100);
camera.position.z = 5;

// ─── Ambient + Point lights ───
scene.add(new THREE.AmbientLight(0x6C63FF, 0.8));
const pointLight = new THREE.PointLight(0x00D4FF, 2, 20);
pointLight.position.set(3, 3, 3);
scene.add(pointLight);

// ─── Geometry objects ───
const objects = [];
const geos = [
  new THREE.IcosahedronGeometry(1, 1),
  new THREE.TorusGeometry(0.8, 0.3, 16, 100),
  new THREE.OctahedronGeometry(0.9),
  new THREE.TorusKnotGeometry(0.6, 0.2, 100, 16),
];
geos.forEach((geo, i) => {
  const mat = new THREE.MeshStandardMaterial({
    color: i % 2 === 0 ? 0x6C63FF : 0x00D4FF,
    wireframe: true, transparent: true, opacity: 0.6,
  });
  const mesh = new THREE.Mesh(geo, mat);
  mesh.position.set((i - 1.5) * 3, Math.sin(i) * 2, -i * 1.5);
  scene.add(mesh);
  objects.push(mesh);
});

// ─── Star field ───
const starGeo = new THREE.BufferGeometry();
const positions = new Float32Array(6000);
for (let i = 0; i < 6000; i++) positions[i] = (Math.random() - 0.5) * 100;
starGeo.setAttribute('position', new THREE.BufferAttribute(positions, 3));
const starMat  = new THREE.PointsMaterial({ color: 0xffffff, size: 0.05 });
scene.add(new THREE.Points(starGeo, starMat));

// ─── Mouse parallax ───
let mouseX = 0, mouseY = 0;
document.addEventListener('mousemove', e => {
  mouseX = (e.clientX / window.innerWidth  - 0.5) * 2;
  mouseY = (e.clientY / window.innerHeight - 0.5) * 2;
});

// ─── Scroll camera drift ───
let scrollY = 0;
window.addEventListener('scroll', () => scrollY = window.scrollY);

// ─── Render loop ───
const clock = new THREE.Clock();
function animate() {
  requestAnimationFrame(animate);
  const t = clock.getElapsedTime();
  objects.forEach((obj, i) => {
    obj.rotation.x = t * 0.3 * (i % 2 === 0 ? 1 : -1);
    obj.rotation.y = t * 0.5;
    obj.position.y = Math.sin(t + i) * 0.5;
  });
  camera.position.x += (mouseX * 0.5 - camera.position.x) * 0.05;
  camera.position.y += (-mouseY * 0.5 - camera.position.y) * 0.05;
  camera.position.z = 5 - scrollY * 0.004;
  renderer.render(scene, camera);
}
animate();

// ─── Resize handler ───
window.addEventListener('resize', () => {
  camera.aspect = window.innerWidth / window.innerHeight;
  camera.updateProjectionMatrix();
  renderer.setSize(window.innerWidth, window.innerHeight);
});


6. GSAP Scroll Animations (scroll-animations.js)
GSAP ScrollTrigger is the industry standard for scroll animations. These animations fire perfectly in sync with the user's scroll position. Paste into js/scroll-animations.js.
// ─── js/scroll-animations.js ─────────────────────────
gsap.registerPlugin(ScrollTrigger);

// ─── Locomotive Scroll + GSAP sync ───
const locoScroll = new LocomotiveScroll({
  el: document.querySelector('[data-scroll-container]'),
  smooth: true, multiplier: 1, class: 'is-inview'
});
locoScroll.on('scroll', ScrollTrigger.update);
ScrollTrigger.scrollerProxy('[data-scroll-container]', {
  scrollTop(value) {
    return arguments.length ? locoScroll.scrollTo(value, 0, 0) : locoScroll.scroll.instance.scroll.y;
  },
  getBoundingClientRect() {
    return { top:0, left:0, width: window.innerWidth, height: window.innerHeight };
  },
  pinType: document.querySelector('[data-scroll-container]').style.transform ? 'transform' : 'fixed'
});

// ─── Section headings: slide up & fade ───
gsap.utils.toArray('.section-title').forEach(el => {
  gsap.from(el, {
    scrollTrigger: { trigger: el, scroller: '[data-scroll-container]', start: 'top 85%', toggleActions: 'play none none reverse' },
    y: 60, opacity: 0, duration: 1, ease: 'power3.out'
  });
});

// ─── Project cards: stagger cascade ───
gsap.utils.toArray('.project-card').forEach((card, i) => {
  gsap.from(card, {
    scrollTrigger: { trigger: card, scroller: '[data-scroll-container]', start: 'top 90%', toggleActions: 'play none none reverse' },
    y: 80, opacity: 0, rotateX: 15, duration: 0.8, delay: i * 0.1, ease: 'power2.out'
  });
});

// ─── Skill pills: pop in ───
gsap.from('.skill-pill', {
  scrollTrigger: { trigger: '#skills', scroller: '[data-scroll-container]', start: 'top 80%' },
  scale: 0, opacity: 0, duration: 0.5, stagger: 0.06, ease: 'back.out(1.7)'
});

// ─── Hero text: staggered entrance ───
gsap.from(['.hero-greeting','.hero-name','.hero-role','.hero-bio','.hero-cta'], {
  y: 40, opacity: 0, duration: 1, stagger: 0.15, ease: 'power3.out', delay: 0.2
});

// ─── Parallax background images ───
gsap.utils.toArray('.parallax-bg').forEach(bg => {
  gsap.to(bg, {
    scrollTrigger: { trigger: bg.parentElement, scroller: '[data-scroll-container]',
      start: 'top bottom', end: 'bottom top', scrub: 1.5 },
    y: '-25%', ease: 'none'
  });
});

ScrollTrigger.addEventListener('refresh', () => locoScroll.update());
ScrollTrigger.refresh();

    size:    { value: 2, random: true, anim: { enable:true, speed:2, minimum:0.5, sync:false } },
    links:   { enable:true, distance:120, color:'#6C63FF', opacity:0.3, width:1 },

14. Performance Optimisation Tips
A beautiful 3D site that loads slowly will hurt your impression. Follow these rules to keep it fast.
Compress all images with Squoosh (free): https://squoosh.app — target <200KB per image
Add loading='lazy' attribute to all <img> tags below the fold
Use renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2)) to limit GPU load on retina displays
Reduce particle count to 60-80 on mobile by detecting screen width before tsParticles.load()
Use IntersectionObserver to pause Three.js animation when not in viewport
Add <link rel='preload'> for your hero background image in the <head>
Enable Gzip/Brotli compression on Netlify via a netlify.toml config file
Use WebP image format instead of JPEG/PNG — 30-50% smaller file sizes
